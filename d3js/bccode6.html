Setting up y-axis


var y = d3.scale.sqrt()
          .domain([0, 300])
          .range([height, 0]);

var yAxis = d3.svg.axis()
              .scale(y)
              .orient("left");

svg.append("g")
   .attr("class", "y axis")
   .call(yAxis)
 .append('text')
   .attr('dx', 5)
   .attr('dy', 10)
   .text('number of medals');
