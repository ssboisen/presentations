Setting up x-axis


var x = d3.scale.linear()
          .domain([15, 50]) //age
          .range([0, width]);

var xAxis = d3.svg.axis()
              .scale(x)
              .orient('bottom');

svg.append('g')
   .attr('transform', 'translate(' + 0 + ',' + height + ')')
   .attr('class', 'x axis')
   .call(xAxis)
 .append('text')
   .attr('x', width)
   .attr('text-anchor', 'end')
   .attr('dy', -5)
   .text('age');
