Scales

A function mapping from data domain to visual range


var r = d3.scale.linear()
          .domain(d3.extent(dataset))
          .rangeRound([25, 100]);

var color = d3.scale.linear()
              .domain(d3.extent(dataset))
              .range('green', 'black')

d3.selectAll('circle')
  .style('r', r)
  .style('fill', color);
