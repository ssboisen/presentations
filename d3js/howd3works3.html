Enter, Update and Exit

Lifecycle of an element bound to data


// update
var p = d3.select('body')
  .selectAll('p')
  .data(newData)  // elements that already in the DOM
  .text(function(d) { return d + ' has been here all along'; });

// enter
p.enter()  // elements that are about to enter the DOM
  .append('p')
  .text(function(d) { return d + ' is new here'; });

// exit
p.exit()  // elements that are about to exit the DOM
  .remove();
