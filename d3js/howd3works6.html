Transitions

Custom attribute tweening


d3.selectAll('span.pctComplete')
  .data([100])
  .text("0%")
  .transition()
    .duration(2000)
    .tween("text", function(d) {
        var i = d3.interpolate(this.textContent, d);
        return function(t) {
            this.textContent = Math.round(i(t)) + "%"
        };});
