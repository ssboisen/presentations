D3s Fluent Syntax


var dataset = [20, 5, 10, 11, 50];

d3.select('body')
  .selectAll('div')  // selection of virtual nodes
  .data(dataset)  // data binding
  .enter()
    .append('div')  // dom manipulation
      .style('height', "10px") //Static attribute
      .style('margin-bottom', '1px')
      .style('width', //dynamic attribute
        function (d) { return d + "px"; })
      .style('background-color', 'black');
